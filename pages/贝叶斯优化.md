- API调用
	- ```python
	  # ==========================================
	  # 1. 定义 Optuna 目标函数（自动搜索最优超参）
	  # ==========================================
	  def objective(trial):
	      # 定义超参数搜索空间（你可以自己加参数）
	      params = {
	          "n_estimators": trial.suggest_int("n_estimators", 50, 300),
	          "max_depth": trial.suggest_int("max_depth", 3, 12),
	          "min_samples_split": trial.suggest_int("min_samples_split", 2, 10),
	          "random_state": 42,
	          "n_jobs": -1
	      }
	  
	      # 模型
	      model = RandomForestClassifier(**params)
	  
	      # 分层10折交叉验证（企业分类任务必用）
	      skf = StratifiedKFold(n_splits=10, shuffle=True, random_state=42)
	      score = cross_val_score(model, X_train_val, y_train_val, cv=skf, scoring="accuracy").mean()
	  
	      return score  # 返回要优化的指标
	  
	  # ==========================================
	  # 2. 启动 Optuna 搜索（企业标准：50~100次迭代）
	  # ==========================================
	  study = optuna.create_study(direction="maximize", study_name="ML Model Tuning")
	  study.optimize(objective, n_trials=50)  # 50次智能搜索，比网格搜索快10倍
	  
	  # 输出最优超参数
	  best_params = study.best_params
	  print("\n✅ Optuna 找到最优超参数：")
	  print(best_params)
	  ```
